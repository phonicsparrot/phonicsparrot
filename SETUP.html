Teacher Setup Guide

Complete guide to running and managing Phonics Parrot in your classroom.

🚀 Quick Start

  • 1
    Extract the filesExtract the downloaded .7z or .zip file onto your desktop or computer folder.
  • 2
    Start the local serverDouble-click the START.bat script file in the extracted folder.
  • 3
    Open Phonics ParrotA browser window will automatically launch to http://localhost:3000.
  • 4
    Keep the server runningMinimise the black console terminal window that popped up. It must remain open while students are playing.
💡
Manual URL: If the web browser doesn't open automatically, just launch Chrome or Microsoft Edge and navigate directly to http://localhost:3000.
📡
Classroom Display & Tablets: Other devices on the same Wi-Fi network (such as tablets or smart screens) can access the app at the network address printed in the terminal (e.g., http://10.x.x.x:3000).

☁️ Google Drive & Sheets Integration

Connecting Phonics Parrot to Google Drive enables student voice recordings to upload directly into your cloud folders, and automatically logs performance metrics to a Google Sheet.

Step 1: Set up Google Apps Script

  • 1
    Open Apps ScriptNavigate to your browser and open Google Apps Script, then click New Project.
  • 2
    Clean the fileDelete the default starter code inside the script window.
  • 3
    Copy & modifyCopy the AppScript code provided on the right using the copy button, then paste it in.
  • 4
    Link SpreadsheetReplace the value of SPREADSHEET_URL at the very top of the script with your real Google Sheet URL, then save with Ctrl+S.

Step 2: Deploy as a Web App

  • 5
    Initiate DeploymentClick the blue Deploy button in the top right corner, then select New deployment.
  • 6
    Configure Web AppClick the gear icon (⚙️) next to "Select type", choose Web app, and set Execute as: Me & Who has access: Anyone.
  • 7
    Copy Script URLClick Deploy, authorize permissions, and copy the generated Web App URL.

Step 3: Connect to Phonics Parrot

  • 8
    Edit config fileOpen drive_config.json in your app folder using Notepad.
  • 9
    Paste URLReplace the placeholder URL inside with your Web App URL, save the file, and restart `START.bat`.
appsscript/Code.gs 
var SPREADSHEET_URL = "https://docs.google.com/spreadsheets/d/YOUR_SPREADSHEET_ID_HERE/edit";
var RECORDINGS_FOLDER_NAME = "Phonics Parrot Recordings";

var COL = { TIMESTAMP: 1, STUDENT: 2, ACTIVITY: 3, TARGET: 4, PHONEME: 5, SPOKEN: 6, SCORE: 7, VERDICT: 8, RECORDING: 9 };
var TOTAL_COLS = 9;

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    if (data.audioData) {
      var driveUrl = saveAudioToDrive(data);
      var cacheKey = getCacheKey(data.class, data.student);
      try { CacheService.getScriptCache().put(cacheKey, driveUrl, 120); } catch (_) {}
      var patchStatus = patchLastRowDriveUrl(data.class, data.student, driveUrl);
      return respond({ result: "success", driveUrl: driveUrl, patchStatus: patchStatus });
    }
    logToSheet(data);
    return respond({ result: "success" });
  } catch (err) {
    return respond({ result: "error", message: err.toString() });
  }
}

function respond(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function getCacheKey(className, studentName) {
  var c = (className || "guest").trim().toLowerCase().replace(/[^a-z0-9]/g, "_");
  var s = (studentName || "guest").trim().toLowerCase().replace(/[^a-z0-9]/g, "_");
  return "pp_rec_" + c + "_" + s;
}

function saveAudioToDrive(data) {
  var audioBytes = Utilities.base64Decode(data.audioData);
  var filename = data.filename || ((data.student || "guest") + "_recording.webm");
  var blob = Utilities.newBlob(audioBytes, "audio/webm", filename);

  var myDrive = DriveApp.getRootFolder();
  var rootFolders = myDrive.getFoldersByName(RECORDINGS_FOLDER_NAME);
  var rootFolder = rootFolders.hasNext() ? rootFolders.next() : myDrive.createFolder(RECORDINGS_FOLDER_NAME);

  var className = (data.class || "Guest").replace(/[\/\\:*?"<>|]/g, "_");
  var studentName = (data.student || "Guest").replace(/[\/\\:*?"<>|]/g, "_");
  var activity = (data.activity || "activity").replace(/[\/\\:*?"<>|]/g, "_");

  var classFolder = getOrCreateSubfolder(rootFolder, className);
  var studentFolder = getOrCreateSubfolder(classFolder, studentName);
  var actFolder = getOrCreateSubfolder(studentFolder, activity);

  var file = actFolder.createFile(blob);
  try { file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch (e) {}
  return file.getUrl();
}

function getOrCreateSubfolder(parent, name) {
  var iter = parent.getFoldersByName(name);
  return iter.hasNext() ? iter.next() : parent.createFolder(name);
}

function patchLastRowDriveUrl(className, studentName, driveUrl) {
  if (!driveUrl) return "No driveUrl provided";
  try {
    var ss = SpreadsheetApp.openByUrl(SPREADSHEET_URL);
    var targetStudent = (studentName || "").trim().toLowerCase();

    if (className) {
      var sheet = ss.getSheetByName(className.trim());
      if (sheet && patchSheetRow(sheet, targetStudent, driveUrl)) {
        return "Patched in class tab: " + className;
      }
    }

    var sheets = ss.getSheets();
    for (var s = 0; s < sheets.length; s++) {
      if (patchSheetRow(sheets[s], targetStudent, driveUrl)) {
        return "Patched in tab: " + sheets[s].getName();
      }
    }
    return "No unlinked row found across any tab";
  } catch (err) {
    return "Error: " + err.toString();
  }
}

function patchSheetRow(sheet, targetStudent, driveUrl) {
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) return false;

  var scanStart = Math.max(2, lastRow - 15);
  var numRows = lastRow - scanStart + 1;
  var values = sheet.getRange(scanStart, 1, numRows, TOTAL_COLS).getValues();

  for (var i = values.length - 1; i >= 0; i--) {
    var rowStudent = String(values[i][COL.STUDENT - 1] || "").trim().toLowerCase();
    var rowRecLink = String(values[i][COL.RECORDING - 1] || "").trim();
    var isMatch = (rowStudent === targetStudent) || (!targetStudent || targetStudent === "guest");

    if (isMatch && !rowRecLink) {
      var targetRow = scanStart + i;
      var cell = sheet.getRange(targetRow, COL.RECORDING);
      try { cell.setFormula('=HYPERLINK("' + driveUrl + '","🎙 Listen")'); } catch (e) { cell.setValue(driveUrl); }
      cell.setFontColor("#1155CC").setHorizontalAlignment("center");
      return true;
    }
  }
  return false;
}

function logToSheet(data) {
  var ss = SpreadsheetApp.openByUrl(SPREADSHEET_URL);
  var tabName = (data.class || data.teacher || "Guest").trim();

  var sheet = ss.getSheetByName(tabName);
  if (!sheet) {
    sheet = ss.insertSheet(tabName);
    setupClassSheet(sheet, tabName);
  }

  var score = (data.score !== undefined && data.score !== "") ? Number(data.score) : "";
  var verdict = data.verdict || getVerdict(score);
  var newRow = sheet.getLastRow() + 1;

  var cacheKey = getCacheKey(data.class || data.teacher, data.student);
  var cachedDriveUrl = "";
  try {
    cachedDriveUrl = CacheService.getScriptCache().get(cacheKey) || "";
    if (cachedDriveUrl) CacheService.getScriptCache().remove(cacheKey);
  } catch (_) {}

  sheet.appendRow([
    data.timestamp || new Date().toISOString(),
    data.student || "Guest",
    formatActivity(data.activity),
    data.target || "",
    data.phoneme || "",
    data.spoken || "",
    score,
    verdict,
    ""
  ]);

  if (cachedDriveUrl) {
    var recCell = sheet.getRange(newRow, COL.RECORDING);
    try { recCell.setFormula('=HYPERLINK("' + cachedDriveUrl + '","🎙 Listen")'); } catch (e) { recCell.setValue(cachedDriveUrl); }
    recCell.setFontColor("#1155CC").setHorizontalAlignment("center");
  }

  if (score !== "") colorScoreCell(sheet.getRange(newRow, COL.SCORE), score);
  if (verdict) colorVerdictCell(sheet.getRange(newRow, COL.VERDICT), score);
  if (newRow % 2 === 0) sheet.getRange(newRow, 1, 1, TOTAL_COLS).setBackground("#F3F6FF");
  sheet.getRange(newRow, COL.SPOKEN).setWrap(true);
}

function formatActivity(activity) {
  var map = { "builder": "🔡 Flashcard Builder", "reader": "📖 Line Reader", "speak": "🗣️ Poem Builder", "pingpong": "🏓 Phonics Pong" };
  return map[activity] || (activity || "");
}

function getVerdict(score) {
  if (score === "" || score === undefined || score === null) return "";
  if (score >= 80) return "🌟 Excellent";
  if (score >= 50) return "👍 Good";
  return "💪 Keep Trying";
}

function colorScoreCell(cell, score) {
  if (score >= 80) cell.setBackground("#C9EAC9").setFontColor("#1B5E20").setFontWeight("bold");
  else if (score >= 50) cell.setBackground("#FFF9C4").setFontColor("#6D4C00").setFontWeight("bold");
  else cell.setBackground("#FFCDD2").setFontColor("#B71C1C").setFontWeight("bold");
  cell.setHorizontalAlignment("center");
}

function colorVerdictCell(cell, score) {
  if (score >= 80) cell.setBackground("#E8F5E9").setFontColor("#2E7D32");
  else if (score >= 50) cell.setBackground("#FFFDE7").setFontColor("#F57F17");
  else cell.setBackground("#FFEBEE").setFontColor("#C62828");
}

function setupClassSheet(sheet, className) {
  sheet.insertRowBefore(1);
  var titleRange = sheet.getRange(1, 1, 1, TOTAL_COLS);
  titleRange.merge().setValue("📚  Class: " + className + "   —   Phonics Parrot Performance Log")
    .setBackground("#1A237E").setFontColor("#FFFFFF").setFontWeight("bold").setFontSize(14).setHorizontalAlignment("center").setVerticalAlignment("middle");
  sheet.setRowHeight(1, 44);

  var headers = ["Timestamp", "Student", "Activity", "Target Phrase", "Phoneme", "Spoken", "Score (%)", "Verdict", "Recording"];
  sheet.appendRow(headers);

  var headerRange = sheet.getRange(2, 1, 1, TOTAL_COLS);
  headerRange.setBackground("#283593").setFontColor("#FFFFFF").setFontWeight("bold").setFontSize(10).setHorizontalAlignment("center").setVerticalAlignment("middle");
  sheet.setRowHeight(2, 32);

  sheet.setColumnWidth(COL.TIMESTAMP, 175); sheet.setColumnWidth(COL.STUDENT, 130); sheet.setColumnWidth(COL.ACTIVITY, 160);
  sheet.setColumnWidth(COL.TARGET, 210); sheet.setColumnWidth(COL.PHONEME, 90); sheet.setColumnWidth(COL.SPOKEN, 210);
  sheet.setColumnWidth(COL.SCORE, 80); sheet.setColumnWidth(COL.VERDICT, 135); sheet.setColumnWidth(COL.RECORDING, 110);
  sheet.setFrozenRows(2);

  var palette = ["#E53935","#8E24AA","#1E88E5","#00897B","#43A047","#F4511E","#6D4C41","#546E7A","#3949AB","#00ACC1"];
  var hash = 0;
  for (var i = 0; i < className.length; i++) hash = (hash * 31 + className.charCodeAt(i)) & 0xFFFF;
  sheet.setTabColor(palette[hash % palette.length]);
}

💾 Local Storage & Safety

Even if you choose not to set up the Google Cloud Integration, your student records are fully saved locally:

  • Audio recordings are stored directly in the recordings/ folder.
  • Grades and progress logs are written to performance_log.json.

This means you can always view student work locally, even without an active internet connection.

❓ Frequently Asked Questions

Does the application require internet?
Yes, but only for the speech-to-text microphone capability (which queries the browser's speech recognition engine). The interactive lessons and data storage are fully local and offline.
Can multiple tablets connect to a single host computer?
Yes. Simply open the URL displayed in the server terminal (e.g. http://192.168.x.x:3000) on all tablets connected to the same Wi-Fi.
What happens if a port already in use error occurs?
Phonics Parrot will automatically detect the port conflict and bind to the next available port (e.g. 3001). Check the terminal output to verify the port number.
Which web browsers are supported?
We recommend using Google Chrome or Microsoft Edge, as Firefox has limited Speech Recognition support.
